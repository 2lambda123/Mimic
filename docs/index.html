function applyFilter() { var allRows = document.querySelectorAll('table tr'); var checkboxes = document.querySelectorAll('.row-checkbox'); // Hide all rows except the header allRows.forEach(function(row, index) { if(index === 0) return; // Skip the header row row.style.display = 'none'; }); // Show only rows with checked checkboxes checkboxes.forEach(function(checkbox, index) { if(checkbox.checked) { allRows[index + 1].style.display = ''; // +1 to account for the header row } }); }
